161     /**
162      * Add the job to Hudson, if necessary.
163      * 
164      * @throws MojoExecutionException if something goes wrong
165      */
166     public void execute() throws MojoExecutionException {
167         if (project == null) {
168             throw new MojoExecutionException("This mojo must be run in the context of a project.");
169         }
170         createHudsonURL();
171 
172         setupClient();
173 
174         if (!jobExists()) {
175             URL postURL = createPostURL();
176             getLog().info("posting job to " + postURL);
177 
178             String payload = createPayload();
179 
180             postPayload(postURL, payload);
181         }
182     }

184     /**
185      * Create a paylod XML for submission to Hudson.
186      * 
187      * @return the XML to post to Hudson
188      * @throws MojoExecutionException if something goes wrong
189      */
190     protected String createPayload() throws MojoExecutionException {
191         try {
192             Velocity.init();
193             VelocityContext ctx = new VelocityContext();
194             ctx.put("goals", createFullGoals());
195             ctx.put("subversionURL", subversionURL.replace("scm:svn:", ""));
196             addIncludes(ctx);
197             if (allocatePorts != null) {
198                 ctx.put("ports", allocatePorts.split(","));
199             }
200             ctx.put("startVNC", Boolean.valueOf(startVNC));
201             if (description != null) {
202                 ctx.put("description", description);
203             } else {
204                 ctx.put("description", "");
205             }
206             StringWriter sw = new StringWriter();
207             Velocity.evaluate(ctx, sw, "project-template.vm", getClass().getResourceAsStream(
208                     "/project-template.vm"));
209             return sw.toString();
210         } catch (Exception e) {
211             throw new MojoExecutionException("Unable to evaluate template", e);
212         }
213     }

246      * 
247      * @return true if the job already exists in Hudson
248      * @throws MojoExecutionException
249      */
250     protected boolean jobExists() throws MojoExecutionException {
251         String jobURL = String.format("%s/api/xml?xpath=/hudson/job[name='%s']", hudsonURL,
252                 getJobName());
253         GetMethod gm = new GetMethod(jobURL);
254         try {
255             int status = httpClient.executeMethod(gm);
256             return status != HttpStatus.SC_NOT_FOUND;
257         } catch (IOException e) {
258             throw new MojoExecutionException("Unable to check if job currently exists", e);
259         }
260     }

277     /**
278      * Iterate through the fields for this class and see if they start with
279      * include. If they do, add each one to the context.
280      * 
281      * @param ctx the VelocityContext to add includes to.
282      * @throws IllegalAccessException if something goes wrong in reflection
283      */
284     private void addIncludes(VelocityContext ctx) throws IllegalAccessException {
285         Field[] fields = AddJobMojo.class.getDeclaredFields();
286         for (int i = 0; i < fields.length; i++) {
287             Field field = fields[i];
288             if (field.getName().startsWith("include")) {
289                 ctx.put(field.getName(), Boolean.valueOf(field.getBoolean(this)));
290             }
291         }
292     }

294     /**
295      * Create a String containing all the goals for a build. This includes
296      * whatever goals are specified in the goals property and any goals required
297      * by any enabled plugins.
298      * 
299      * @return a String containing all the goals necessary
300      * @throws MojoExecutionException if something goes wrong in reflection
301      */
302     private String createFullGoals() throws MojoExecutionException {
303         try {
304             StringBuilder builder = new StringBuilder();
305             for (String plugin : PLUGINS) {
306                 if (isPluginEnabled(plugin) && PREFIXES.containsKey(plugin)) {
307                     builder.append(StringUtils.join(PREFIXES.get(plugin).toArray(), " "));
308                     builder.append(" ");
309                 }
310             }
311 
312             if (performClean) {
313                 builder.append("clean ");
314             }
315 
316             builder.append(primaryGoal);
317             builder.append(" ");
318 
319             for (String plugin : PLUGINS) {
320                 if (isPluginEnabled(plugin) && GOALS.containsKey(plugin)) {
321                     builder.append(StringUtils.join(GOALS.get(plugin).toArray(), " "));
322                     builder.append(" ");
323                 }
324             }
325 
326             if (additionalGoals != null) {
327                 builder.append(additionalGoals);
328             }
329 
330             return builder.toString().trim();
331         } catch (Exception e) {
332             throw new MojoExecutionException("Exception while creating goals", e);
333         }
334     }

338      * upper cased) and return the value.
339      * 
340      * @param plugin the plugin name
341      * @return if the plugin is enabled
342      * @throws NoSuchFieldException if the field does not exist

345     private boolean isPluginEnabled(String plugin) throws NoSuchFieldException,
346             IllegalAccessException {
347         String fieldName = "include" + plugin.substring(0, 1).toUpperCase() + plugin.substring(1);
348         Field field = AddJobMojo.class.getDeclaredField(fieldName);
349         return field.getBoolean(this);
