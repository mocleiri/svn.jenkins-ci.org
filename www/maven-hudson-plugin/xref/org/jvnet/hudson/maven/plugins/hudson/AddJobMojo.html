178      * Add the job to Hudson, if necessary.

182     public void execute() throws MojoExecutionException {
183         if (project == null) {
184             throw new MojoExecutionException("This mojo must be run in the context of a project.");
185         }
186         createHudsonURL();
187 
188         setupClient();
189 
190         if (!jobExists()) {
191             URL postURL = createPostURL();
192             getLog().info("posting job to " + postURL);

194             String payload = createPayload();
195 
196             postPayload(postURL, payload);

200     /**
201      * Create a paylod XML for submission to Hudson.
202      * 
203      * @return the XML to post to Hudson
204      * @throws MojoExecutionException if something goes wrong
205      */
206     protected String createPayload() throws MojoExecutionException {
207         try {
208             Velocity.init();
209             VelocityContext ctx = new VelocityContext();
210             ctx.put("goals", createFullGoals());
211             ctx.put("subversionURL", subversionURL.replace("scm:svn:", ""));
212             addIncludes(ctx);
213             if (allocatePorts != null) {
214                 ctx.put("ports", allocatePorts.split(","));
215             }
216             ctx.put("startVNC", Boolean.valueOf(startVNC));
217             if (description != null) {
218                 ctx.put("description", description);
219             } else {
220                 ctx.put("description", "");
221             }
222             StringWriter sw = new StringWriter();
223             Velocity.evaluate(ctx, sw, "project-template.vm", getClass().getResourceAsStream(
224                     "/project-template.vm"));
225             return sw.toString();
226         } catch (Exception e) {
227             throw new MojoExecutionException("Unable to evaluate template", e);
228         }
229     }

258     /**
259      * Method to check if a job already exists.

261      * @param hudsonURL The Base Hudson URL
262      * 
263      * @return true if the job already exists in Hudson
264      * @throws MojoExecutionException
265      */
266     protected boolean jobExists() throws MojoExecutionException {
267         String jobURL = String.format("%s/api/xml?xpath=/hudson/job[name='%s']", hudsonURL,
268                 getJobName());
269         GetMethod gm = new GetMethod(jobURL);
270         try {
271             int status = httpClient.executeMethod(gm);
272             return status != HttpStatus.SC_NOT_FOUND;
273         } catch (IOException e) {
274             throw new MojoExecutionException("Unable to check if job currently exists", e);
275         }
276     }

297      * @param ctx the VelocityContext to add includes to.
298      * @throws IllegalAccessException if something goes wrong in reflection

300     private void addIncludes(VelocityContext ctx) throws IllegalAccessException {
301         Field[] fields = AddJobMojo.class.getDeclaredFields();
302         for (int i = 0; i < fields.length; i++) {
303             Field field = fields[i];
304             if (field.getName().startsWith("include")) {
305                 ctx.put(field.getName(), Boolean.valueOf(field.getBoolean(this)));
306             }
307         }

310     /**
311      * Create a String containing all the goals for a build. This includes
312      * whatever goals are specified in the goals property and any goals required
313      * by any enabled plugins.
314      * 
315      * @return a String containing all the goals necessary
316      * @throws MojoExecutionException if something goes wrong in reflection
317      */
318     private String createFullGoals() throws MojoExecutionException {
319         try {
320             StringBuilder builder = new StringBuilder();
321             if (shouldFailAtEnd) {
322                 builder.append("-fae ");
323             }
324 
325             if (shouldUseBatchMode) {
326                 builder.append("-B ");
327             }
328 
329             for (String plugin : PLUGINS) {
330                 if (isPluginEnabled(plugin) && PREFIXES.containsKey(plugin)) {
331                     builder.append(StringUtils.join(PREFIXES.get(plugin).toArray(), " "));
332                     builder.append(" ");
333                 }
334             }
335 
336             if (performClean) {
337                 builder.append("clean ");
338             }
339 
340             builder.append(primaryGoal);
341             builder.append(" ");
342 
343             for (String plugin : PLUGINS) {
344                 if (isPluginEnabled(plugin) && GOALS.containsKey(plugin)) {
345                     builder.append(StringUtils.join(GOALS.get(plugin).toArray(), " "));
346                     builder.append(" ");
347                 }
348             }
349 
350             if (additionalGoals != null) {
351                 builder.append(additionalGoals);
352             }
353 
354             return builder.toString().trim();
355         } catch (Exception e) {
356             throw new MojoExecutionException("Exception while creating goals", e);
357         }
358     }
