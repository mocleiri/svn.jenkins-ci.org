82      protected void configurePlugin(Document jobDoc, PluginType pluginType, Element element) {
83          Element plugins = jobDoc.getRootElement().getChild(pluginType.name());
84          if (plugins.getChild(element.getName()) != null) {
85              plugins.removeChild(element.getName());
86          }
87          plugins.addContent(element);

90      /**
91       * Create the base URL for Hudson. Will first look at the ciManagement
92       * section of the pom and then override that with the hudson.url property
93       * (if available).
94       * 
95       * @return the Base URL for hudson (without a trailing slash)
96       * @throws MojoExecutionException if there's no hudson url
97       * @throws MalformedURLException if the hudson url is badly formed
98       */
99      protected void createHudsonURL() throws MojoExecutionException {
100         try {
101             if (hudsonURL == null) {
102                 if ((project != null) && (project.getCiManagement() != null)
103                         && ("hudson".equals(project.getCiManagement().getSystem()))) {
104                     hudsonURL = new URL(project.getCiManagement().getUrl());
105                 }
106             }
107 
108             if (hudsonURL == null) {
109                 throw new MojoExecutionException("No hudson url defined.");
110             }
111 
112             String hudsonURLAsString = hudsonURL.toString();
113             if (hudsonURLAsString.endsWith("/")) {
114                 hudsonURLAsString = hudsonURLAsString.substring(0, hudsonURLAsString.length() - 1);
115                 hudsonURL = new URL(hudsonURLAsString);
116             }
117         } catch (MalformedURLException e) {
118             throw new MojoExecutionException("Unable to create URL", e);
119         }
120 
121         getLog().info("Base Hudson URL is " + hudsonURL);
122 
123     }

215     /**
216      * Make an HTTP Post to the URL, erroring out if the returned status doesn't
217      * make an expectation.
218      * 
219      * @param url the URL to POST to.
220      * @param requestEntity an entity to include with the POST.
221      * @param expectedStatus expected HTTP status code.
222      * @throws IOException if something goes wrong
223      * @throws MojoExecutionException if the status doesn't match
224      */
225     protected void doPost(URL url, RequestEntity requestEntity, int expectedStatus)
226             throws IOException, MojoExecutionException {
227         PostMethod pm = new PostMethod(url.toString());
228         if (requestEntity != null) {
229             pm.setRequestEntity(requestEntity);
230         }
231         int status = httpClient.executeMethod(pm);
232         if (status != expectedStatus) {
233             throw new MojoExecutionException("Incorrect status back from Hudson: " + status
234                     + ". Expected: " + expectedStatus);
235         }
236     }

238     /**
239      * Search the Maven Settings object for a Server object that matches the URL
240      * provided. If the port isn't provided or is the default, then the Server
241      * ID is expected to be just the hostname. If the port is provided and is
242      * not the default, the Server ID is hostname:port.
243      * 
244      * @return the corresponding Server object or null
245      */
246     protected Server findServer() {
247         if (settings == null) {
248             return null;
249         }
250         String serverID = hudsonURL.getHost();
251         if ((hudsonURL.getPort() != -1) && (hudsonURL.getPort() != hudsonURL.getDefaultPort())) {
252             serverID = serverID + ":" + hudsonURL.getPort();
253         }
254         return settings.getServer(serverID);
255     }

263     protected void forEachJob(JobClosure closure) throws MojoExecutionException {
264         List<String> jobList = null;
265         try {
266             jobList = getJobList();
267         } catch (IOException e) {
268             throw new MojoExecutionException("Unable to obtain job list", e);
269         }
270 
271         for (String job : jobList) {
272             closure.doWithJob(job);
273 
274         }

277     protected Document getDocumentFromHudson(URL url) throws MojoExecutionException {
278         try {
279             GetMethod gm = new GetMethod(url.toString());
280             int status = httpClient.executeMethod(gm);
281             if (status != HttpStatus.SC_OK) {
282                 throw new MojoExecutionException("Non-OK status back from Hudson: " + status);
283             }
284 
285             return new SAXBuilder().build(gm.getResponseBodyAsStream());
286         } catch (JDOMException e) {
287             throw new MojoExecutionException("Unable to parse " + url.toString(), e);
288         } catch (IOException e) {
289             throw new MojoExecutionException("Unable to parse " + url.toString(), e);
290         }
291     }

293     /**
294      * Fetch the job list from Hudson.
295      * 
296      * @throws MojoExecutionException
297      * @throws IOException
298      */
299     @SuppressWarnings("unchecked")
300     protected List<String> getJobList() throws MojoExecutionException, IOException {
301         URL jobListURL = createJobListURL();
302         Document doc = getDocumentFromHudson(jobListURL);
303 
304         List<Element> nodes = doc.getRootElement().getChildren("name");
305         List<String> retval = new ArrayList<String>(nodes.size());
306         for (Element node : nodes) {
307             retval.add(node.getTextNormalize());
308         }
309 
310         return retval;
311     }

320      * 
321      * @param job the job name

323      */
324     protected void rebuildJob(String job) throws MojoExecutionException {
325         URL buildURL = createJobBuildURL(job);
326         try {
327             doPost(buildURL, HttpStatus.SC_MOVED_TEMPORARILY);
328         } catch (HttpException e) {
329             throw new MojoExecutionException("Unable to rebuild job " + job, e);
330         } catch (IOException e) {
331             throw new MojoExecutionException("Unable to rebuild job " + job, e);
332         }

338      * @param name the job name
339      * @param jobDoc the job config document

342     protected void saveJobConfig(String name, Document jobDoc) throws MojoExecutionException {
343         URL configURL = createJobConfigURL(name);
344         String payload = xmlOutputter.outputString(jobDoc);
345         try {
346             doPost(configURL, new StringRequestEntity(payload, "text/xml", "UTF-8"),
347                     HttpStatus.SC_OK);
348         } catch (IOException e) {
349             throw new MojoExecutionException("Exception posting payload to " + configURL, e);
350         }

354     /**
355      * Configure the HttpClient instance with any necessary authentication
356      * parameters.
357      * 
358      * @param client
359      * @param hudsonURL
360      */
361     protected void setupClient() {
362         Server server = findServer();
363         if (server != null) {
364             httpClient.getParams().setAuthenticationPreemptive(true);
365             Credentials defaultcreds = new UsernamePasswordCredentials(server.getUsername(), server
366                     .getPassword());
367             httpClient.getState().setCredentials(
368                     new AuthScope(hudsonURL.getHost(), hudsonURL.getPort(), AuthScope.ANY_REALM),
369                     defaultcreds);
370         }
371     }
