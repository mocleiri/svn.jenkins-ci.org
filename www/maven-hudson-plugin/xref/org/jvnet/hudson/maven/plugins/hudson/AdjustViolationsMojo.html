22  public class AdjustViolationsMojo extends AbstractHudsonMojo {
23  
24      /**
25       * The job name.
26       * 
27       * @parameter expression="${jobName}"
28       */
29      private String jobName;
30  
31      /**
32       * Maximum number of violations for Cloudy status.
33       * 
34       * @parameter expression="${max}"
35       * @required
36       */
37      private int max;
38  
39      /**
40       * Minimum number of violations for Cloudy status.
41       * 
42       * @parameter expression="${min}"
43       * @required
44       */
45      private int min;
46  
47      /**
48       * Violation number at which the build becomes unstable.
49       * 
50       * @parameter expression="${unstable}"
51       * @required
52       */
53      private int unstable;
54  
55      public void execute() throws MojoExecutionException, MojoFailureException {
56          setupClient();
57          if (jobName != null) {
58              changeJob(jobName);
59              rebuildJob(jobName);
60          } else {
61              throw new MojoExecutionException("No job name specified");
62          }
63  
64      }
65  
66      protected void changeJob(String name) throws MojoExecutionException {
67          getLog().info("Modifying job: " + name);
68          Document jobDoc = loadJobConfig(name);
69          try {
70              changeViolations(jobDoc);
71          } catch (JDOMException e) {
72              throw new MojoExecutionException("Unable to change violations.");
73          }
74          saveJobConfig(name, jobDoc);
75      }
76  
77      /**
78       * Change the configuration of the Violations plugin with the appropriate
79       * min, max, and unstable values.
80       * 
81       * @param jobDoc
82       *            the job Document
83       * @return true if the violations Element was changed
84       * @throws JDOMException
85       *             if the
86       */
87      @SuppressWarnings("unchecked")
88      protected boolean changeViolations(Document jobDoc) throws JDOMException {
89          List<Element> typeConfigs = XPath
90                  .selectNodes(
91                          jobDoc,
92                          "/project/publishers/hudson.plugins.violations.ViolationsPublisher/config/typeConfigs/entry/hudson.plugins.violations.TypeConfig");
93          if (typeConfigs.isEmpty()) {
94              return false;
95          } else {
96              for (Element typeConfig : typeConfigs) {
97                  createOrChange(typeConfig, "min", min);
98                  createOrChange(typeConfig, "max", max);
99                  createOrChange(typeConfig, "unstable", unstable);
100             }
101             return true;
102         }
